while (x == 'Y' || x == 'y')

    cin >> y;
    age = 2025 - y;
    cout << age;
    cout << "Are you want to continue (Y/N)?";
    cin >> x;
}
// (ملاحظة: لو كنت كتبته من غير خطوة 1 مكنش هيتنفذ خطوة 2)
